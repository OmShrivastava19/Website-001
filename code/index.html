Agree Or Not

I Made A Program To Print Hello World!



// Logical operators
#include <cs50.h>
#include <stdio.h>

int main(void)
{
	// Prompt user to agree
	char c = get_char("Do you agree?\n");
	
	// Check whether agreed
	if (c == 'Y' || c == 'y')
	{
		printf("Agreed.\n");
	}
	else if (c == 'N' || c == 'n')
	{
		printf("Not agreed.\n");
	}
}

Condition

I Made A Program To Print Hello World!



// Conditions and relational operators
#include <cs50.h>
#include <stdio.h>

int main(void)
{
	// Prompt user for x
	int x = get_int("x: ");

	// Prompt user for y
	int y = get_int("y: ");

	// Compare x and y
	if (x < y)
	{
		printf("x is less than y\n");
	}
	else if (x > y)
	{
		printf("x is greater than y\n");
	}
	else
	{
		printf("x is equal to y\n");
	}
}
